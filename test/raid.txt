herramientas

particionar sdb
# fdisk /dev/sdb

nueva particion
-> n
particion primaria
-> p
sectores
-> \n * 3
tipo de particion
-> t
tipo RAID
-> fd
configuracion realizada
-> p
escribir particion
-> w

creacion del raid1
# mdadm -C /dev/md0 -l raid1 -n 2 /dev/sd[b-c]1

comando del pfr
# mdadm --create --verbose /dev/md0 --level=raid1 --raid-device=2 /dev/sdb1Â /dev/sdc1

#sudo reboot

verificar RAID
# mdadm --detail /dev/md0
# mdadm --detail /dev/md127

comando pfr
# cat /proc/mdstat

formatear volumen del RAID
# mkfs.ext4 /dev/md0
# mkfs.ext4 /dev/md127

creacion directorio
# mkdir /media/raid

propietario del raid
# chown chrlr: /media/raid

mount -t ext4 /dev/md0 /media/raid
